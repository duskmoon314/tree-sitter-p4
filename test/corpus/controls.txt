control MyIngress(inout headers_t hdr) {
    apply {
        if (hdr.ethernet.ether_type == 16w0x0800) {
            hdr.ethernet.dst_addr = hdr.ethernet.src_addr;
        }
    }
}

            hdr.ethernet.dst_addr: exact;

            forward;
            drop;
        }
    }
    apply {
        my_table.apply();
    }
